\titleformat{\section}%
  [hang]% shape
  {\sffamily\LARGE\color{myGreen}}% format applied to label+text
  {}% label
  {1em}% horizontal separation between label and title body
  {\thesection\hspace{0.3em}}% before the title body
  []% after the title body

  {\thesubsection\hspace{0.3em}}% before the title body
  []% after the title body
